UE = CS.UnityEngine

velocity = 5

liveTime = 6
m_liveTime = 0

pipe_rb = nil

function LuaStart()
    --print("Running pipe_start")
    pipe_rb = CS.Extensions.AddRigidbody2D(self.gameObject)
    pipe_rb.gravityScale = 0.0

    pipe_rb.velocity = UE.Vector2.left * velocity
end

function LuaUpdate()
    --print("Running pipe_update")
    
    m_liveTime = m_liveTime + UE.Time.deltaTime

    if (m_liveTime >= liveTime) then
        --print("Đã tới lúc Destroy pipe")

        m_liveTime = 0
        UE.Object.Destroy(self.gameObject)
    end
end


function GetVector3D(x, y, z)
    return UE.Vector3(x, y, z)
end

function GetVector2D(x, y)
    return UE.Vector2(x, y)
end
